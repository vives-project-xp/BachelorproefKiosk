#include "Keyboard.h"
#include "Mouse.h"

// the setup function runs once when you press reset or power the board
void setup() {
  pinMode(3,INPUT_PULLUP);    //up // Joystick_Gray
  pinMode(4,INPUT_PULLUP);    //right // Joystick_White
  pinMode(5,INPUT_PULLUP);    //down // Joy_Orange
  pinMode(6,INPUT_PULLUP);    //left // Joystick_Brown
  pinMode(7,INPUT_PULLUP);    //knop 1 // Buttom_Gray
  pinMode(8,INPUT_PULLUP);    //knop 2 // Buttom_Purple
  pinMode(9,OUTPUT);          //low output voor elke pheriferal // RED
  digitalWrite(9, LOW);
  Keyboard.begin();
  Mouse.begin();
}

void loop() {
  if(digitalRead(8)==LOW){
    Keyboard.press(KEY_LEFT_ALT);
    Keyboard.press(KEY_LEFT_ARROW);
  }else{
    Keyboard.release(KEY_LEFT_ALT);
    Keyboard.release(KEY_LEFT_ARROW);
  }
  if(digitalRead(7)==LOW){
    Mouse.press(MOUSE_LEFT);
  }else{
    Mouse.release(MOUSE_LEFT);
  }
  if(digitalRead(6)==LOW){
    Mouse.move(-3, 0);
  }
  if(digitalRead(5)==LOW){
        Mouse.move(0, 3);
  }
  if(digitalRead(4)==LOW){
    Mouse.move(3, 0);
  }
  if(digitalRead(3)==LOW){
        Mouse.move(0, -3);
  }
}
